#include <iostream>

using namespace std;

int main() {
    double a, b, res;
    int operacion;
    do {
        cout << "Operaciones basicas\n" << endl;
        cout << "---------------------" << endl;
        cout << "1. Suma" << endl;
        cout << "2. Resta" << endl;
        cout << "3. Multiplicacion" << endl;
        cout << "4. Division" << endl;
        cout << "0. Salir" << endl;
        cout << "---------------------" << endl;
        cout << "Ingrese la opcion que desea realizar: ";
        cin >> operacion;

        switch (operacion) {
            case 1:
                cout << "Operacion 1: SUMA" << endl;
                cout << "Proporcione el valor de a: ";
                cin >> a;
                cout << "Proporcione el valor de b: ";
                cin >> b;
                res = a + b;
                cout << a << " + " << b << " = " << res << endl;
                break;
            case 2:
                cout << "Operacion 2: RESTA" << endl;
                cout << "Proporcione el valor de a: ";
                cin >> a;
                cout << "Proporcione el valor de b: ";
                cin >> b;
                res = a - b;
                cout << a << " - " << b << " = " << res << endl;
                break;
            case 3:
                cout << "Operacion 3: MULTIPLICACION" << endl;
                cout << "Proporcione el valor de a: ";
                cin >> a;
                cout << "Proporcione el valor de b: ";
                cin >> b;
                res = a * b;
                cout << a << " * " << b << " = " << res << endl;
                break;
            case 4:
                cout << "Operacion 4: DIVISION" << endl;
                cout << "Proporcione el valor de a: ";
                cin >> a;
                cout << "Proporcione el valor de b: ";
                cin >> b;
                if (b != 0) {
                    res = a / b;
                    cout << a << " / " << b << " = " << res << endl;
                } else {
                    cout << "Error: No se puede dividir entre 0" << endl;
                }
                break;
            case 0:
                cout << "Saliendo del programa..." << endl;
                break;
            default:
                cout << "Operacion no valida" << endl;
                break;
        }
        cout << endl;
    } while (operacion != 0);

    system("pause");
    return 0;
}
